// Remove Duplicates
public class add {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String s = "Programming";
        // op -> "Progamin"
        int n = s.length();
        for(int i = 0 ; i < n ; i++){
            // for checking if it already came
            boolean flag = true;
            for(int j = 0 ; j < i ; j++){
                if(s.charAt(i) == s.charAt(j)){
                    flag = false;
                    break;
                }
            }
            if(flag)
                System.out.print(s.charAt(i));
        }
    }
}
